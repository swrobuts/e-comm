import pandas as pd # CSV laden df = pd.read_csv('reviews.csv') # Erste Zeilen anzeigen print(df.head()) # Spalten: review_id, customer_id, product_name, rating, review_text, sentiment # Statistik print(df['rating'].describe()) # count: 5857, mean: ~3.7, min: 1, max: 5 # Sentiment-Verteilung print(df['sentiment'].value_counts()) # positive: 3982, neutral: 1003, negative: 872

from vaderSentiment.vaderSentiment import SentimentIntensityAnalyzer analyzer = SentimentIntensityAnalyzer() # Score fuer jeden Review berechnen def get_sentiment(text): if pd.isna(text): return 0 scores = analyzer.polarity_scores(str(text)) return scores['compound'] # -1 bis +1 df['vader_score'] = df['review_text'].apply(get_sentiment) # Klassifizierung def classify(score): if score >= 0.05: return 'positive' elif score <= -0.05: return 'negative' else: return 'neutral' df['vader_label'] = df['vader_score'].apply(classify) # Vergleich: vorklassifiziert vs. VADER print(df[['sentiment', 'vader_label']].value_counts())

import plotly.express as px # 1. Rating-Histogramm fig1 = px.histogram(df, x='rating', nbins=5, color='rating', color_discrete_map={1:'#DC2626', 2:'#F97316', 3:'#9CA3AF', 4:'#E87722', 5:'#3D7A56'}, title='UrbanCart: Rating-Verteilung', labels={'rating': 'Sterne', 'count': 'Anzahl Reviews'}) fig1.show() # 2. Sentiment pro Produkt (100%-Balken) sent_pct = df.groupby(['product_name', 'sentiment']).size().reset_index(name='count') fig2 = px.bar(sent_pct, x='product_name', y='count', color='sentiment', color_discrete_map={'positive':'#3D7A56', 'neutral':'#9CA3AF', 'negative':'#DC2626'}, barmode='relative', title='Sentiment-Verteilung pro Produkt') fig2.update_layout(xaxis_tickangle=45) fig2.show()

from wordcloud import WordCloud, STOPWORDS import matplotlib.pyplot as plt # Negative Reviews filtern neg_reviews = df[df['sentiment'] == 'negative']['review_text'].dropna() # Text zusammenfuehren neg_text = ' '.join(neg_reviews.tolist()) # Stopwoerter entfernen (Deutsch + Englisch) stopwords = set(STOPWORDS) stopwords.update(['das', 'die', 'der', 'und', 'ist', 'ich', 'es', 'ein', 'nicht', 'sehr', 'aber', 'auch', 'war']) # Wordcloud erstellen wc = WordCloud( width=800, height=400, background_color='white', stopwords=stopwords, colormap='Reds', max_words=50 ).generate(neg_text) plt.figure(figsize=(14, 7)) plt.imshow(wc, interpolation='bilinear') plt.axis('off') plt.title('Haeufige Begriffe in negativen Reviews', fontsize=16) plt.tight_layout() plt.savefig('wordcloud_negativ.png', dpi=150) plt.show() print("Top-Themen in negativen Reviews:") word_freq = wc.words_ for word, freq in sorted(word_freq.items(), key=lambda x: -x[1])[:10]: print(f" {word}: {freq:.3f}")
